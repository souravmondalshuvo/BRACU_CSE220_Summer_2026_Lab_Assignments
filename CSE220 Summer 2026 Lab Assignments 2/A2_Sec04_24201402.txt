// AssignmentTask1: Building Blocks

    public static String checkSimilar( Node building1, Node building2 ){

        Node a = building1;
        Node b = building2;

        while(a != null && b != null) {
            if(!a.elem.equals(b.elem)) {
                return "Not Similar";
            }

            a = a.next;
            b = b.next;
        }

        if(a == null && b == null) {
            return "Similar";
        } else {
            return "Not Similar";
        }
    }

// AssignmentTask2: Organize Books

    public static Node organizeBooks(Node head, Integer[] popularity) {

        if(head == null) {
            return null;
        }

        boolean kono_kichu_swapped_hoise = true;

        while(kono_kichu_swapped_hoise) {
            kono_kichu_swapped_hoise = false;
            
            Node current = head;
            int index = 0;

            while(current.next != null) {
                if(popularity[index] < popularity[index + 1]) {
                    String title_hold_kore = (String)current.elem;

                    current.elem = current.next.elem;
                    current.next.elem = title_hold_kore;

                    Integer score_hold_kore = popularity[index];

                    popularity[index] = popularity[index + 1];
                    popularity[index + 1] = score_hold_kore;

                    kono_kichu_swapped_hoise = true;
                }

                current = current.next;
                index++;
            }
        }

        return head;
    }

// AssignmentTask3: Alternate Merge

    public static Node alternateMerge( Node head1, Node head2 ){

        Node walk1 = head1;
        Node walk2 = head2;

        while(walk1 != null && walk2 != null) {
            Node rest1 = walk1.next;
            Node rest2 = walk2.next;

            walk1.next = walk2;

            if(rest1 != null) {
                walk2.next = rest1;
            }

            walk1 = rest1;
            walk2 = rest2;
        }

        return head1;
    }

// AssignmentTask4: ID Generator

    public static Node idGenerator(Node head1, Node head2, Node head3) {

        Node id_head = null;
        Node cursor = head1;

        while(cursor != null) {
            id_head = new Node((int) cursor.elem, id_head);
            cursor = cursor.next;
        }

        Node tail = id_head;

        while(tail != null && tail.next != null) {
            tail = tail.next;
        }

        Node p2 = head2;
        Node p3 = head3;

        while(p2 != null && p3 != null) {
            int sob_eksathe = (int) p2.elem + (int) p3.elem;
            sob_eksathe = sob_eksathe % 10;

            Node banailo = new Node(sob_eksathe, null);

            tail.next = banailo;
            tail = banailo;
            p2 = p2.next;
            p3 = p3.next;
        }
        
        return id_head;
    }
