// AssignmentTask1: Hashtable: Deletion Operation

    private int hashFunction( Integer key ){

        return Math.floorMod(key + 3, ht.length);

    }

    public void remove( Integer key ){

        int index = hashFunction(key);

        PairNode bortoman = ht[index];
        PairNode otit = null;

        while(bortoman != null) {
            if(bortoman.key.equals(key)) {
                if(otit == null) {
                    ht[index] = bortoman.next;
                } else {
                    otit.next = bortoman.next;
                }

                return;
            }

            otit = bortoman;
            bortoman = bortoman.next;
        }
    }

// AssignmentTask2: Searching in Hashtable

    private int hashFunction( String key ){

        String s = key;

        if(s.length() % 2 != 0) {
            s = s + "N";
        }

        int sum = 0;

        for(int i = 0; i < s.length(); i += 2) {
            int prothom = (int) s.charAt(i);
            int ditiyo = (int) s.charAt(i + 1);
            String eksathe = "" + prothom + ditiyo;

            sum += Integer.parseInt(eksathe);
        }

        return sum % ht.length;

    }

    public String searchHashtable( Object[] keyValuePair ){

        String key = (String) keyValuePair[0];
        int index = hashFunction(key);

        PairNode node = ht[index];

        while(node != null) {
            if(node.key.equals(key)) {
                return "Found";
            }

            node = node.next;
        }

        return "Not Found";

    }
