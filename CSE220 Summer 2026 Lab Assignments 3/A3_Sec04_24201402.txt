    private int hashFunction( Integer key ){

        return Math.floorMod(key + 3, ht.length);

    }

    public void remove( Integer key ){

        int index = hashFunction(key);

        PairNode bortoman = ht[index];
        PairNode otit = null;

        while(bortoman != null) {
            if(bortoman.key.equals(key)) {
                if(otit == null) {
                    ht[index] = bortoman.next;
                } else {
                    otit.next = bortoman.next;
                }

                return;
            }

            otit = bortoman;
            bortoman = bortoman.next;
        }
    }

// AssignmentTask2: Searching in Hashtable

    private int hashFunction( String key ){

        String s = key;

        if(s.length() % 2 != 0) {
            s = s + "N";
        }

        int sum = 0;

        for(int i = 0; i < s.length(); i += 2) {
            int prothom = (int) s.charAt(i);
            int ditiyo = (int) s.charAt(i + 1);
            String eksathe = "" + prothom + ditiyo;

            sum += Integer.parseInt(eksathe);
        }

        return sum % ht.length;

    }

    public String searchHashtable( Object[] keyValuePair ){

        String key = (String) keyValuePair[0];
        int index = hashFunction(key);

        PairNode node = ht[index];

        while(node != null) {
            if(node.key.equals(key)) {
                return "Found";
            }

            node = node.next;
        }

        return "Not Found";

    }

// AssignmentTask1: Stack: Conditional Reverse

    public static Stack conditionalReverse(Stack stack) {

        Stack result = new Stack();
        Integer aage = null;

        while(!stack.isEmpty()) {
            int bortoman = stack.pop();

            if(aage == null || bortoman != aage) {
                result.push(bortoman);
            }

            aage = bortoman;
        }

        return result;
    }

// AssignmentTask2: Queue: Hot Potato Game

    public static String hotPotato(String[] players, int k) {

        LinkedListQueue q = new LinkedListQueue();

        for(int i = 0; i < players.length; i++) {
            q.enqueue(players[i]);
        }

        int baaki_players = players.length;

        while(baaki_players > 1) {
            for(int i = 0; i < k; i++) {
                q.enqueue(q.dequeue());
            }

            String baad = (String) q.dequeue();
            System.out.println(baad + " elimminated");
            baaki_players--;
        }

        return (String) q.dequeue();
    }

// AssignmentTask3: Stack and Queue: Dancing Pairs

    public static void dance_pair(Stack stk) {

        Queue cheleder_q = new Queue();
        Queue meyeder_q = new Queue();

        int count = 0;

        while(!stk.isEmpty()) {
            Dancer prothom = (Dancer) stk.pop();

            if(stk.isEmpty()) {
                if(prothom.getGender() == 'M') {
                    cheleder_q.enqueue(prothom);
                } else {
                    meyeder_q.enqueue(prothom);
                }

                break;
            }

            Dancer ditiyo = (Dancer) stk.pop();

            char g1 = prothom.getGender();
            char g2 = ditiyo.getGender();

            if(g1 != g2) {
                count++;
                System.out.println("#" + count + ": " + prothom + " & " + ditiyo);
            } else if(g1 == 'F') {
                meyeder_q.enqueue(prothom);
                stk.push(ditiyo);
            } else {
                if(!meyeder_q.isEmpty()) {
                    Dancer waiting_meye = (Dancer) meyeder_q.dequeue();
                    count++;
                    System.out.println("#" + count + ": " + prothom + " & " + waiting_meye);
                } else {
                    cheleder_q.enqueue(prothom);
                }

                stk.push(ditiyo);
            }
        }

        while(!cheleder_q.isEmpty() && !meyeder_q.isEmpty()) {
            Dancer m = (Dancer) cheleder_q.dequeue();
            Dancer f = (Dancer) meyeder_q.dequeue();
            count++;
            System.out.println("#" + count + ": " + m + " & " + f);
        }

        if(!cheleder_q.isEmpty()) {
            System.out.println("Unpaired Male:");

            String line = "";

            while(!cheleder_q.isEmpty()) {
                line += cheleder_q.dequeue();
                if(!cheleder_q.isEmpty()) {
                    line += ", ";
                }

                System.out.println(line);
            }
        }

        if(!meyeder_q.isEmpty()) {
            System.out.println("Unpaired Female:");

            String line = "";

            while(!meyeder_q.isEmpty()) {
                line += meyeder_q.dequeue();
                if(!meyeder_q.isEmpty()) {
                    line += ", ";
                }

                System.out.println(line);
            }
        }

    }
